94  private async void OnMessage(object sender, MessageEventArgs messageEventArgs)
95  {
96  Message message = messageEventArgs.Message;
97  IMessage adaptedMessage = new TelegramMessageAdapter(message);
98  string chatID = adaptedMessage.ChatID;
99 
100  // Responde los mensajes que se recibieron mientras estaba caído el bot de forma amigable.
101  if (message.Date < this.startupTime)
102  {
103  await this.TelegramBot.Cliente.SendTextMessageAsync(adaptedMessage.ChatID, "Perdona, no estaba disponible.");
104  return;
105  }
106 
107  if (adaptedMessage.TxtMensaje == "/start")
108  {
109  this.CrearConversacion(adaptedMessage.ChatID);
110  }
111 
112  string frmPrevioMensaje;
113  string respuesta;
114  string frmPostMensaje;
115  do
116  {
117  frmPrevioMensaje = this.ObtenerFormularioActual(chatID).GetType().ToString();
118  respuesta = this.ObtenerFormularioActual(chatID).ReceiveMessage(adaptedMessage);
119  frmPostMensaje = this.ObtenerFormularioActual(chatID).GetType().ToString();
120  }
121  while (frmPrevioMensaje != frmPostMensaje);
122 
123  await this.TelegramBot.Cliente.SendTextMessageAsync(adaptedMessage.ChatID, respuesta);
124  }
125  }
