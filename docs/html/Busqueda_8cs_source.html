78  public List<Publicacion> BuscarPublicaciones(Dictionary<FiltrosPosibles, object> publicacionesASeparar)
79  {
80  if (publicacionesASeparar == null)
81  {
82  throw new ArgumentNullException(nameof(publicacionesASeparar), "publicacionesASeparar es null");
83  }
84 
85  List<Publicacion> result = new List<Publicacion>();
86  List<Publicacion> publicacionesNoAptas = new List<Publicacion>();
87  List<Publicacion> publicacionesActivas = da.Obtener<Publicacion>()
88  .Concat(da.Obtener<PublicacionRecurrente>())
89  .Where((Publicacion p) => !p.Comprado).ToList();
90 
91  foreach (var filtro in publicacionesASeparar)
92  {
93  foreach (Publicacion suspect in publicacionesActivas)
94  {
95  // Con el transcurso del curso, nos hemos dado cuenta de que este enfoque a través
96  // de un diccionario y un switch es frágil. En retrospectiva, y con miras a implementarlo
97  // para la entrega final, consideramos que podríamos crear varias clases que implementen
98  // una interfaz común IFiltroBusqueda, de forma que cada filtro concreto implemente su
99  // acción de filtro correspondiente.
100  switch (filtro.Key)
101  {
102  case FiltrosPosibles.Empresa:
103  if (!suspect.Vendedor.Equals(filtro.Value))
104  {
105  publicacionesNoAptas.Add(suspect);
106  }
107 
108  break;
109  case FiltrosPosibles.Residuo:
110  if (!suspect.Residuo.Equals(filtro.Value))
111  {
112  publicacionesNoAptas.Add(suspect);
113  }
114 
115  break;
116  case FiltrosPosibles.LugarRetiro:
117  if (!suspect.LugarRetiro.Equals(filtro.Value))
118  {
119  publicacionesNoAptas.Add(suspect);
120  }
121 
122  break;
123  case FiltrosPosibles.PrecioMaximo:
124  if (suspect.PrecioUnitario > (double)filtro.Value)
125  {
126  publicacionesNoAptas.Add(suspect);
127  }
128 
129  break;
130  case FiltrosPosibles.FrecuenciaRestock:
131  if ((int) filtro.Value == 0)
132  {
133  if (suspect is PublicacionRecurrente)
134  {
135  publicacionesNoAptas.Add(suspect);
136  }
137  }
138  else
139  {
140  if (suspect is not PublicacionRecurrente || (suspect as PublicacionRecurrente).FrecuenciaAnualRestock != (int) filtro.Value)
141  {
142  publicacionesNoAptas.Add(suspect);
143  }
144  }
145  break;
146  case FiltrosPosibles.Categorias:
147  if (suspect.Residuo.Categoria != filtro.Value)
148  {
149  publicacionesNoAptas.Add(suspect);
150  }
151  break;
152  default:
153  break;
154  }
155  }
156  }
157 
158  foreach (Publicacion publicacion in publicacionesActivas)
159  {
160  if (!publicacionesNoAptas.Contains(publicacion))
161  {
162  result.Add(publicacion);
163  }
164  }
165 
166  return result;
167  }
168  }
